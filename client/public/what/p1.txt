        protected void Button2_Click(object sender, EventArgs e)

            int n, i, s = 0; 
            n = Convert.ToInt32(TextBox1.Text); 
            if (n == 0 || n == 1) 
                s = 1; 
            for (i = 2; i <= n / 2; ++i) 
            { 
                if (n % i == 0) 
                { 
                    s = 1; 
                    break; 
                } 
            } 
            if (s == 0) 
                Label3.Text = "The given number is prime"; 
            else 
                Label3.Text = "The given number is not prime"; 
        }
