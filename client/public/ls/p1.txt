Using One LED

import RPI.GPIO as GPIO
import time
GPIO.setmode (GPIO.BOARD)
GPIO.setup(15,GPIO.OUT)
for i in range(5):
GPIO.output(15,True)
print(&quot;ON&quot;)
time.sleep(1)
GPIO.output(15,False)
print(&quot;OFF&quot;)
time.sleep(1)
print(&quot;Done&quot;)
GPIO.cleanup()
